loadstring(game:HttpGet("https://raw.githubusercontent.com/daucobonhi/Ui-Redz-V2/refs/heads/main/UiREDzV2.lua"))()

local Window = MakeWindow({
    Hub = {
        Title = "SPEED HACK",
        Color = Color3.fromRGB(255, 0, 0)
    }
})

local currentSpeed = 16
local enabled = false
local humanoid

local function updateSpeed()
    if game.Players.LocalPlayer.Character then
        humanoid = game.Players.LocalPlayer.Character:FindFirstChild("Humanoid")
        if humanoid then
            humanoid.WalkSpeed = enabled and currentSpeed or 16
        end
    end
end

AddSlider(MakeTab({Name = "Speed"}), {
    Name = "Tốc độ (0-9999)",
    Min = 0,
    Max = 9999,
    Default = 16,
    Callback = function(value)
        currentSpeed = value
        if enabled then updateSpeed() end
    end
})

AddToggle(MakeTab({Name = "Speed"}), {
    Name = "Bật Speed Hack",
    Default = false,
    Callback = function(state)
        enabled = state
        updateSpeed()
    end
})

game.Players.LocalPlayer.CharacterAdded:Connect(function()
    if enabled then updateSpeed() end
end)
